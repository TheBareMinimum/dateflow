<script>

let map;

document.getElementById('findBtn').addEventListener('click', findCafes);

async function findCafes() {

const area = document.getElementById('area').value;

const budget = document.getElementById('budget').value;

const vibe = document.getElementById('vibe').value;

const radiusKm = document.getElementById('radius').value;

const [lat, lng] = area.split(',');

document.getElementById('loader').style.display = 'block';

document.getElementById('results').style.display = 'none';

document.getElementById('map').style.display = 'none';

document.getElementById('error').style.display = 'none';

document.getElementById('findBtn').disabled = true;

try {

const params = new URLSearchParams({query:vibe,ll:lat+','+lng,radius:Math.round(radiusKm*1000),max_price:budget,min_price:1,limit:15,sort:'RATING'});

const res = await fetch('/.netlify/functions/cafes?' + params.toString());

if (!res.ok) throw new Error('Function returned ' + res.status + '. ' + (await res.text()).slice(0,200));

const data = await res.json();

if (!data.results || !data.results.length) throw new Error("No cafés found. Try wider radius or different mood.");

const normalized = data.results.map(p=>({name:p.name,rating:p.rating,price:p.price,categories:p.categories||[],location:p.location||{},lat:p.latitude||(p.geocodes&&p.geocodes.main&&p.geocodes.main.latitude),lng:p.longitude||(p.geocodes&&p.geocodes.main&&p.geocodes.main.longitude)})).filter(p=>p.lat&&p.lng);

const top3 = normalized.sort((a,b)=>(b.rating||0)-(a.rating||0)).slice(0,3);

if (!top3.length) throw new Error("Could not map results.");

renderMap(lat, lng, top3);

renderCards(top3);

} catch (err) {

document.getElementById('error').innerHTML = err.message + '<small>V2.0 · If you see 404, the Netlify Function did not deploy. Use GitHub route.</small>';

document.getElementById('error').style.display = 'block';

} finally {

document.getElementById('loader').style.display = 'none';

document.getElementById('findBtn').disabled = false;

}

}

function renderMap(lat,lng,cafes){document.getElementById('map').style.display='block';if(map)map.remove();map=L.map('map',{scrollWheelZoom:false}).setView([lat,lng],14);L.tileLayer('https://{s}.basemaps.cartocdn.com/rastertiles/voyager/{z}/{x}/{y}{r}.png',{attribution:'© OSM © CARTO',maxZoom:19}).addTo(map);const colors=['#D4A5A5','#D9C5A0','#B8C5B0'];const bounds=[];cafes.forEach((c,i)=>{const pos=[c.lat,c.lng];bounds.push(pos);const icon=L.divIcon({html:'<div style="background:'+colors[i]+';color:#3A302B;width:32px;height:32px;border-radius:50%;display:flex;align-items:center;justify-content:center;font-weight:500;font-family:Cormorant Garamond,serif;box-shadow:0 4px 14px rgba(58,48,43,0.25);border:2px solid white;">'+(i+1)+'</div>',className:'',iconSize:[32,32],iconAnchor:[16,16]});L.marker(pos,{icon}).addTo(map).bindPopup('<b>'+c.name+'</b>'+(c.location.formatted_address||'')+'<br>'+(c.rating?'★ '+c.rating+'/10':''));});if(bounds.length>1)map.fitBounds(bounds,{padding:[60,60]});setTimeout(()=>map.invalidateSize(),100);}

function renderCards(cafes){const container=document.getElementById('cards');container.innerHTML='';['One','Two','Three'].forEach((label,i)=>{const c=cafes[i];if(!c)return;const mapsUrl='https://www.google.com/maps/search/?api=1&query='+encodeURIComponent(c.name+' '+(c.location.locality||'Mumbai'));const cats=(c.categories||[]).slice(0,2).map(x=>'<span class="tag">'+x.name+'</span>').join('');const card=document.createElement('div');card.className='card';card.innerHTML='<div class="card-number">'+label+'</div><h3>'+c.name+'</h3><div class="meta">'+(c.location.formatted_address||'')+'</div>'+(c.price?'<div class="meta">'+'₹'.repeat(c.price)+'</div>':'')+'<div class="tags">'+cats+'</div>'+(c.rating?'<div class="rating">★ '+c.rating+' / 10</div>':'')+'<a href="'+mapsUrl+'" target="_blank" class="maps-link">Open in maps →</a>';container.appendChild(card);});document.getElementById('results').style.display='block';}
